  3. ResilientBClient.java ‚Äî Resilient implementation
  @Component
  @ConditionalOnProperty(name = "resilience.enabled", havingValue = "true")
  public class ResilientBClient implements BClientPort {

      // Injected config
      @Value("${b.service.url}")          String bServiceUrl;
      @Value("${b.deadline.ms:800}")      int deadlineMs;
      @Value("${b.inflight.max:10}")      int maxInflight;
      @Autowired                          MetricsService metricsService;

      // Resilience components
      private Semaphore semaphore;         // bulkhead
      private CircuitBreaker cb;          // fail-fast breaker
      private ManagedChannel channel;
      private DemoServiceGrpc.DemoServiceBlockingStub stub;

      @PostConstruct
      public void init() {
          // 1. gRPC channel (same as BClient)
          channel = ManagedChannelBuilder.forTarget(bServiceUrl).usePlaintext().build();
          stub = DemoServiceGrpc.newBlockingStub(channel);

          // 2. Bulkhead semaphore
          semaphore = new Semaphore(maxInflight, true);  // fair

          // 3. Circuit breaker
          CircuitBreakerConfig config = CircuitBreakerConfig.custom()
              .slidingWindowType(COUNT_BASED)
              .slidingWindowSize(10)
              .minimumNumberOfCalls(5)
              .failureRateThreshold(50.0f)           // open at 50% failures
              .slowCallRateThreshold(80.0f)          // open at 80% slow calls
              .slowCallDurationThreshold(Duration.ofMillis(deadlineMs))
              .waitDurationInOpenState(Duration.ofSeconds(10))
              .permittedNumberOfCallsInHalfOpenState(3)
              .automaticTransitionFromOpenToHalfOpenEnabled(true)
              .build();
          cb = CircuitBreakerRegistry.of(config).circuitBreaker("B");

          // 4. Wire breaker state ‚Üí metrics
          cb.getEventPublisher().onStateTransition(e -> {
              int state = switch (cb.getState()) {
                  case OPEN      -> 1;
                  case HALF_OPEN -> 2;
                  default        -> 0;  // CLOSED, DISABLED, etc.
              };
              metricsService.setBreakerState(state);
          });

          logger.info("ResilientBClient initialized: deadline={}ms, maxInflight={}", deadlineMs, maxInflight);
      }

      @Override
      public WorkResult callWork(String requestId) {
          // Step 1: Bulkhead check (non-blocking tryAcquire)
          if (!semaphore.tryAcquire()) {
              metricsService.recordDownstreamCall(0, ErrorCode.QUEUE_FULL);
              logger.warn("Bulkhead full [inflight={}] ‚Üí QUEUE_FULL", maxInflight);
              return new WorkResult(false, ErrorCode.QUEUE_FULL.name(), 0, ErrorCode.QUEUE_FULL);
          }

          // Step 2: Circuit breaker check
          if (!cb.tryAcquirePermission()) {
              semaphore.release();
              metricsService.recordDownstreamCall(0, ErrorCode.CIRCUIT_OPEN);
              logger.warn("Circuit open ‚Üí CIRCUIT_OPEN");
              return new WorkResult(false, ErrorCode.CIRCUIT_OPEN.name(), 0, ErrorCode.CIRCUIT_OPEN);
          }

          long start = System.currentTimeMillis();
          metricsService.incrementInflight();
          try {
              // Step 3: gRPC call with deadline
              WorkReply reply = stub
                  .withDeadlineAfter(deadlineMs, TimeUnit.MILLISECONDS)
                  .work(WorkRequest.newBuilder().setId(requestId).build());

              long latency = System.currentTimeMillis() - start;
              cb.onSuccess(latency, TimeUnit.MILLISECONDS);
              metricsService.recordDownstreamCall(latency, ErrorCode.SUCCESS);
              return new WorkResult(reply.getOk(), reply.getCode(), latency, ErrorCode.SUCCESS);

          } catch (StatusRuntimeException e) {
              long latency = System.currentTimeMillis() - start;
              ErrorCode code = ErrorCode.fromGrpcStatus(e.getStatus().getCode());
              cb.onError(latency, TimeUnit.MILLISECONDS, e);
              metricsService.recordDownstreamCall(latency, code);
              logger.error("gRPC call failed: {} ‚Üí {}", e.getStatus(), code);
              return new WorkResult(false, code.name(), latency, code);

          } finally {
              metricsService.decrementInflight();
              semaphore.release();
          }
      }

      @PreDestroy
      public void shutdown() {
          if (channel != null) channel.shutdown();
      }
  }

  - R1: Every call MUST have withDeadlineAfter(deadlineMs) ‚Äî no exceptions
  - R2: Bulkhead MUST use tryAcquire() (non-blocking) ‚Äî never acquire() (blocking)

  - ‚ùå Using semaphore.acquire() (blocks thread, defeats the purpose)
